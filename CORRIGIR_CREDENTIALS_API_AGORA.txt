#!/bin/bash
# ========================================
# CORRIGIR CREDENTIALS.JS - USAR API CENTRALIZADA
# ========================================

cd /home/administrador/CorujaMonitor

echo "🔧 Corrigindo Credentials.js para usar API centralizada..."

# Fazer backup
cp frontend/src/components/Credentials.js frontend/src/components/Credentials.js.bak

# Substituir axios.get por api.get
sed -i 's/axios\.get/api.get/g' frontend/src/components/Credentials.js

# Substituir axios.post por api.post
sed -i 's/axios\.post/api.post/g' frontend/src/components/Credentials.js

# Substituir axios.delete por api.delete
sed -i 's/axios\.delete/api.delete/g' frontend/src/components/Credentials.js

# Remover ${API_URL}/api/v1 das URLs (já está no baseURL)
sed -i 's|\${API_URL}/api/v1/credentials/|/credentials/|g' frontend/src/components/Credentials.js
sed -i 's|\${API_URL}/api/v1/tenants/|/tenants/|g' frontend/src/components/Credentials.js
sed -i 's|\${API_URL}/api/v1/servers/|/servers/|g' frontend/src/components/Credentials.js

# Remover backticks e template strings desnecessários
sed -i "s|\`/credentials/\`|'/credentials/'|g" frontend/src/components/Credentials.js
sed -i "s|\`/tenants/\`|'/tenants/'|g" frontend/src/components/Credentials.js
sed -i "s|\`/servers/\`|'/servers/'|g" frontend/src/components/Credentials.js

echo "✅ Correções aplicadas!"
echo ""
echo "📝 Verificando mudanças:"
grep -n "api\." frontend/src/components/Credentials.js | head -10
echo ""

echo "🔄 Reiniciando frontend..."
docker-compose restart frontend
echo ""

echo "⏳ Aguardando 30 segundos..."
sleep 30
echo ""

echo "🧪 Testando frontend:"
curl -I http://localhost:3000
echo ""

echo "✅ PRONTO! Acesse: http://192.168.31.161:3000"
echo "   Vá em Configurações > Credenciais"
echo "   Agora deve funcionar corretamente!"
