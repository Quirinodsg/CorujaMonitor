#!/bin/bash
# ========================================
# DIAGNÓSTICO COMPLETO DO FRONTEND
# ========================================

echo "🔍 PASSO 1: Verificando arquivos necessários..."
echo ""
echo "Dockerfile do frontend:"
ls -la frontend/Dockerfile
echo ""
echo "package.json do frontend:"
ls -la frontend/package.json
echo ""

echo "🐳 PASSO 2: Verificando containers existentes..."
docker ps -a | grep frontend
echo ""

echo "📋 PASSO 3: Verificando docker-compose.yml..."
grep -A 15 "frontend:" docker-compose.yml
echo ""

echo "🔍 PASSO 4: Verificando se há erros no build anterior..."
docker logs coruja-frontend --tail 50 2>&1 || echo "Container não existe ainda"
echo ""

echo "========================================
SOLUÇÃO: Execute os comandos abaixo
========================================

# 1. Parar todos os containers
docker-compose down

# 2. Remover container frontend antigo (se existir)
docker rm -f coruja-frontend 2>/dev/null || true

# 3. Remover imagem frontend antiga (se existir)
docker rmi corujamonitor_frontend 2>/dev/null || true

# 4. Reconstruir e iniciar TODOS os containers
docker-compose up -d --build

# 5. Aguardar 60 segundos
sleep 60

# 6. Verificar status
docker-compose ps

# 7. Ver logs do frontend
docker logs coruja-frontend --tail 50

# 8. Testar acesso
curl -I http://localhost:3000
"
