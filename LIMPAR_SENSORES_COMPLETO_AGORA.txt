SELECT 
    srv.hostname,
    s.id,
    s.name,
    COUNT(m.id) as metricas,
    MAX(m.timestamp) as ultima_metrica,
    CASE 
        WHEN s.id = (
            SELECT s2.id 
            FROM sensors s2 
            LEFT JOIN metrics m2 ON m2.sensor_id = s2.id 
            WHERE s2.server_id = s.server_id 
            AND s2.sensor_type = 'ping' 
            GROUP BY s2.id 
            ORDER BY COUNT(m2.id) DESC, MAX(m2.timestamp) DESC 
            LIMIT 1
        ) THEN 'MANTER' 
        ELSE 'DELETAR' 
    END as acao
FROM servers srv
JOIN sensors s ON s.server_id = srv.id
LEFT JOIN metrics m ON m.sensor_id = s.id
WHERE s.sensor_type = 'ping'
GROUP BY srv.hostname, s.id, s.name, s.server_id
ORDER BY srv.hostname, metricas DESC;
"

# Ver incidentes de PING abertos
docker exec -it coruja-postgres psql -U coruja -d coruja_monitor -c "
SELECT 
    i.id,
    srv.hostname,
    s.name as sensor_name,
    i.status,
    i.severity,
    i.created_at,
    i.resolved_at
FROM incidents i
JOIN sensors s ON i.sensor_id = s.id
JOIN servers srv ON s.server_id = srv.id
WHERE s.sensor_type = 'ping'
AND i.status != 'resolved'
ORDER BY i.created_at DESC;
"

# Fechar todos os incidentes de PING abertos
docker exec -it coruja-postgres psql -U coruja -d coruja_monitor -c "
UPDATE incidents 
SET status = 'resolved', 
    resolved_at = NOW(),
    resolution_notes = 'Fechado automaticamente - sistema de PING migrado para worker'
WHERE sensor_id IN (
    SELECT id FROM sensors WHERE sensor_type = 'ping'
)
AND status != 'resolved';
"
