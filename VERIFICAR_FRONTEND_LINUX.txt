#!/bin/bash
# ========================================
# VERIFICAR E INICIAR FRONTEND NO LINUX
# ========================================

echo "🔍 Verificando status dos containers..."
docker-compose ps
echo ""

echo "📊 Verificando logs do frontend (últimas 30 linhas)..."
docker logs coruja-frontend --tail 30
echo ""

echo "🔄 Reiniciando todos os containers..."
docker-compose restart
echo ""

echo "⏳ Aguardando containers iniciarem (30 segundos)..."
sleep 30
echo ""

echo "🧪 Testando API:"
curl -I http://localhost:8000/health
echo ""

echo "🧪 Testando Frontend:"
curl -I http://localhost:3000
echo ""

echo "📋 Status final dos containers:"
docker-compose ps
echo ""
