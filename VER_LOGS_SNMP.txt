========================================
COMANDOS POWERSHELL - VER LOGS SNMP
========================================

# Ir para o diretório da probe
cd "C:\Program Files\CorujaMonitor\Probe"

# Ver últimas 50 linhas do log
Get-Content probe.log -Tail 50

# Filtrar apenas linhas com SNMP
Get-Content probe.log | Select-String "SNMP"

# Filtrar SNMP + Parsing + Collected
Get-Content probe.log | Select-String "SNMP|Parsing|Collected|debug"

# Ver logs em tempo real (acompanhar)
Get-Content probe.log -Wait -Tail 20

# Ver apenas últimas 100 linhas filtradas
Get-Content probe.log -Tail 100 | Select-String "SNMP|Parsing|Collected"

# Salvar logs filtrados em arquivo
Get-Content probe.log | Select-String "SNMP" | Out-File snmp_logs.txt
