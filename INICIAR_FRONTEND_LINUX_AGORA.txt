#!/bin/bash
# ========================================
# INICIAR FRONTEND NO LINUX - URGENTE
# ========================================

echo "🔍 Verificando se Dockerfile do frontend existe..."
ls -la frontend/Dockerfile
echo ""

echo "🐳 Parando todos os containers..."
docker-compose down
echo ""

echo "🔨 Reconstruindo frontend (pode demorar alguns minutos)..."
docker-compose build frontend
echo ""

echo "🚀 Iniciando TODOS os containers..."
docker-compose up -d
echo ""

echo "⏳ Aguardando containers iniciarem (45 segundos)..."
sleep 45
echo ""

echo "📋 Status dos containers:"
docker-compose ps
echo ""

echo "📊 Logs do frontend (últimas 40 linhas):"
docker logs coruja-frontend --tail 40
echo ""

echo "🧪 Testando API:"
curl -I http://localhost:8000/health
echo ""

echo "🧪 Testando Frontend:"
curl -I http://localhost:3000
echo ""

echo "2. Login: admin@coruja.com / admin123"
echo "3. Vá em Configurações > Credenciais"
echo "4. Adicione credencial WMI para SRVHVSPRD010"
echo ""

echo "   docker logs coruja-frontend -f"
