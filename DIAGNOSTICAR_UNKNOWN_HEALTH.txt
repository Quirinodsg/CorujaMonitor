## Ver quais sensores aparecem como "Desconhecido" no Status de Saúde
## (sensores ativos sem nenhuma métrica)

docker exec -it coruja-postgres psql -U coruja -d coruja_monitor -c "
SELECT s.id, s.name, s.sensor_type, s.is_active, s.is_acknowledged,
       srv.hostname, srv.ip_address,
       (SELECT COUNT(*) FROM metrics m WHERE m.sensor_id = s.id) as metric_count
FROM sensors s
JOIN servers srv ON s.server_id = srv.id
WHERE s.is_active = true
  AND s.is_acknowledged = false
  AND (SELECT COUNT(*) FROM metrics m WHERE m.sensor_id = s.id) = 0
ORDER BY srv.ip_address, s.id;
"
