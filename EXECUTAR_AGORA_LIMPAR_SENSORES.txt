## PASSO 1 — Deletar sensores duplicados do servidor 110 (IDs 44-49)

docker exec -it coruja-postgres psql -U coruja -d coruja_monitor -c "
DELETE FROM metrics WHERE sensor_id IN (44, 45, 46, 47, 48, 49);
DELETE FROM sensor_notes WHERE sensor_id IN (44, 45, 46, 47, 48, 49);
DELETE FROM sensor_breach_history WHERE sensor_id IN (44, 45, 46, 47, 48, 49);
DELETE FROM sensors WHERE id IN (44, 45, 46, 47, 48, 49);
SELECT 'Sensores duplicados deletados: ' || COUNT(*) FROM sensors WHERE id IN (44,45,46,47,48,49);
"

## PASSO 2 — Ver os sensores unknown para identificar os que não têm métricas

docker exec -it coruja-postgres psql -U coruja -d coruja_monitor -c "
SELECT id, name, sensor_type, verification_status,
       (SELECT COUNT(*) FROM metrics m WHERE m.sensor_id = s.id) as metric_count
FROM sensors s
WHERE verification_status = 'unknown'
ORDER BY metric_count, id;
"
